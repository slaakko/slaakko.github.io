405     [nodiscard]
406     public Result<bool> Read<T>(MemoryReader& reader, List<T>& x)
407     {
408         x.Clear();
409         uint n = 0u;
410         Read(reader, n);
411         for (uint i = 0u; i < n; ++i;)
412         {
413             T m;
414             Read(reader, m);
415             x.Add(m);
416         }
417         return Result<bool>(true);
