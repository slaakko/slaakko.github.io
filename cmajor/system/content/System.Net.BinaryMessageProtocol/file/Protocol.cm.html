 10 namespace System.Net.BinaryMessageProtocol
 11 {
 12     public abstract class BinaryMessage
 13     {
 14         public BinaryMessage(uint id_) : id(id_)
 15         {
 16         }
 17         public virtual ~BinaryMessage()
 18         {
 19         }
 20         public abstract uint Length() const;
 21         [nodiscard]
 22         public abstract Result<bool> Write(MemoryWriter& writer);
 23         [nodiscard]
 24         public abstract Result<bool> Read(MemoryReader& reader);
 25         public uint Id() const { return id; }
 26         private uint id;
 27     }

 29     public uint MakeMessageId(ulong hashCode)
 30     {
 31         uint messageId = cast<uint>(1u + (hashCode % cast<ulong>(4294967295u)));
 32         return messageId;
 33     }

 35     public Result<BinaryMessage*> ReadMessage(TcpSocket& socket)
 36     {
 37         int messageLength = 0;
 38         byte[4] buffer;
 39         int offset = 0;
 40         int bytesToReceive = cast<int>(sizeof(messageLength));
 41         Result<int> receiveResult = socket.Receive(&buffer[offset], bytesToReceive);
 42         if (receiveResult.Error())
 43         {
 44             return Result<BinaryMessage*>(ErrorId(receiveResult.GetErrorId()));
 45         }
 46         int bytesReceived = receiveResult.Value();
 47         if (bytesReceived == 0)
 48         {
 49             return Result<BinaryMessage*>();
 50         }
 51         bytesToReceive = bytesToReceive - bytesReceived;
 52         offset = offset + bytesReceived;
 53         while (bytesToReceive > 0)
 54         {
 55             receiveResult = socket.Receive(&buffer[offset], bytesToReceive);
 56             if (receiveResult.Error())
 57             {
 58                 return Result<BinaryMessage*>(ErrorId(receiveResult.GetErrorId()));
 59             }
 60             bytesReceived = receiveResult.Value();
 61             if (bytesReceived == 0)
 62             {
 63                 return Result<BinaryMessage*>();
 64             }
 65             bytesToReceive = bytesToReceive - bytesReceived;
 66             offset = offset + bytesReceived;
 67         }
 68         MemoryReader lengthReader(&buffer[0], sizeof(messageLength));
 69         auto lengthResult = lengthReader.ReadUInt();
 70         if (lengthResult.Error()) return Result<BinaryMessage*>(ErrorId(lengthResult.GetErrorId()));
 71         messageLength = cast<int>(lengthResult.Value());
 72         UniquePtr<byte> mem(cast<byte*>(MemAlloc(messageLength + 4)));
 73         offset = 0;
 74         bytesToReceive = messageLength + 4;
 75         receiveResult = socket.Receive(mem.Get() + offset, bytesToReceive);
 76         if (receiveResult.Error())
 77         {
 78             return Result<BinaryMessage*>(ErrorId(receiveResult.GetErrorId()));
 79         }
 80         bytesReceived = receiveResult.Value();
 81         if (bytesReceived == 0)
 82         {
 83             return Result<BinaryMessage*>();
 84         }
 85         bytesToReceive = bytesToReceive - bytesReceived;
 86         offset = offset + bytesReceived;
 87         while (bytesToReceive > 0)
 88         {
 89             receiveResult = socket.Receive(mem.Get() + offset, bytesToReceive);
 90             if (receiveResult.Error())
 91             {
 92                 return Result<BinaryMessage*>(ErrorId(receiveResult.GetErrorId()));
 93             }
 94             bytesReceived = receiveResult.Value();
 95             if (bytesReceived == 0)
 96             {
 97                 return Result<BinaryMessage*>();
 98             }
 99             bytesToReceive = bytesToReceive - bytesReceived;
100             offset = offset + bytesReceived;
101         }
102         MemoryReader messageReader(mem.Get(), messageLength + 4);
103         auto messageIdResult = messageReader.ReadUInt();
104         if (messageIdResult.Error())
105         {
106             return Result<BinaryMessage*>(ErrorId(messageIdResult.GetErrorId()));
107         }
108         uint messageId = messageIdResult.Value();
109         Result<BinaryMessage*> createResult = MessageRegistry.Instance().Create(messageId);
110         if (createResult.Error())
111         {
112             return Result<BinaryMessage*>(ErrorId(createResult.GetErrorId()));
113         }
114         BinaryMessage* message = createResult.Value();
115         auto readResult = message->Read(messageReader);
116         if (readResult.Error())
117         {
118             return Result<BinaryMessage*>(ErrorId(readResult.GetErrorId()));
119         }
120         return Result<BinaryMessage*>(message);
121     }
122     [nodiscard]
123     public Result<bool> WriteMessage(TcpSocket& socket, BinaryMessage* message)
124     {
125         uint messageLength = message->Length();
126         byte[4] lengthBuffer;
127         MemoryWriter lengthWriter(&lengthBuffer[0], 4);
128         auto result = lengthWriter.Write(messageLength);
129         if (result.Error()) return result;
130         result = socket.SendAll(&lengthBuffer[0], 4);
131         if (result.Error()) return result;
132         uint messageId = message->Id();
133         byte[4] messageIdBuffer;
134         MemoryWriter idWriter(&messageIdBuffer[0], 4);
135         result = idWriter.Write(messageId);
136         if (result.Error()) return result;
137         result = socket.SendAll(&messageIdBuffer[0], 4);
138         if (result.Error()) return result;
139         if (messageLength > 0)
140         {
141             UniquePtr<byte> mem(cast<byte*>(MemAlloc(messageLength)));
142             MemoryWriter messageWriter(mem.Get(), cast<int>(messageLength));
143             result = message->Write(messageWriter);
144             if (result.Error()) return result;
145             result = socket.SendAll(mem.Get(), cast<int>(messageLength));
146             if (result.Error()) return result;
147         }
148         return Result<bool>(true);
149     }
