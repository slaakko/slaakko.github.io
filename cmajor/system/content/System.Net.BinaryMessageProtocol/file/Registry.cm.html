 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Net.BinaryMessageProtocol
10 {
11     public delegate Result<BinaryMessage*> FactoryFunction(uint messageId);
12 
13     public class MessageRegistry
14     {
15         static MessageRegistry() : instance(new MessageRegistry())
16         {
17         }
18         public static MessageRegistry& Instance()
19         {
20             return *instance;
21         }
22         private MessageRegistry()
23         {
24         }
25         [nodiscard]
26         public Result<bool> Register(const string& className, FactoryFunction factoryFunc)
27         {
28             ulong hashCode = GetHashCode(className);
29             uint messageId = MakeMessageId(hashCode);
30             auto it = factoryMap.Find(messageId);
31             if (it == factoryMap.End())
32             {
33                 factoryMap[messageId] = factoryFunc;
34             }
35             else
36             {
37                 int errorId = AllocateError("cmajor.binary.message.protocol: error: message id \'" + 
38                     ToString(messageId) + "\' already in use, please change the message name");
39                 return Result<bool>(ErrorId(errorId));
40             }
41         }
42         public Result<BinaryMessage*> Create(uint messageId)
43         {
44             auto it = factoryMap.Find(messageId);
45             if (it != factoryMap.End())
46             {
47                 FactoryFunction factoryFunction = it->second;
48                 return factoryFunction(messageId);
49             }
50             else
51             {
52                 int errorId = AllocateError("cmajor.binary.message.protocol: error: message id \'" + ToString(messageId) + "\' not registered");
53                 return Result<BinaryMessage*>(ErrorId(errorId));
54             }
55         }
56         private static UniquePtr<MessageRegistry> instance;
57         private Map<uint, FactoryFunction> factoryMap;
58     }
59 
60     public Result<bool> RegisterMessage<T>()
61     {
62         return MessageRegistry.Instance().Register(T.ClassName(), T.Create);
63     }
