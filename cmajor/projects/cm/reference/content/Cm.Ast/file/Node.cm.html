421         public nothrow void Register(NodeType nodeType, NodeCreator* creator)

424         }
425         public Node* CreateNode(NodeType nodeType, const System.Lex.Span& span)
426         {
427             const UniquePtr<NodeCreator>& creator = creators[cast<long>(cast<byte>(nodeType))];
428             if (!creator.IsNull())
429             {
430                 Node* value = creator->CreateNode(span);
431                 if ((value != null))
432                 {
433                     return value;
434                 }
435                 else
436                 {
437                     throw Exception("could not create node");
438                 }
439             }
440             else
441             {
442                 throw Exception("no creator for node type \'" + NodeTypeStr(nodeType) + "\'");
443             }
444         }
445         private List<UniquePtr<NodeCreator>> creators;
446     }
