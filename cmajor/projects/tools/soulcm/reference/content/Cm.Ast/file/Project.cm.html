528         public void ResolveDeclarations()
529         {
530             for (const UniquePtr<ProjectDeclaration>& declaration : declarations)
531             {
532                 bool unknown = false;
533                 switch (declaration->GetDeclarationType())
534                 {
535                     case ProjectDeclarationType.referenceDeclaration:
536                     {
537                         ReferenceDeclaration* referenceDeclaration = cast<ReferenceDeclaration*>(declaration.Get());
538                         string filePath = Path.MakeCanonical(referenceDeclaration->FilePath());
539                         relativeReferencedProjectFilePaths.Add(filePath);
540                         if (Path.IsAbsolute(filePath))
541                         {
542                             referencedProjectFilePaths.Add(GetFullPath(filePath));
543                         }
544                         else
545                         {
546                             string absolutePath = GetFullPath(Path.Combine(sourceBasePath, filePath));
547                             referencedProjectFilePaths.Add(absolutePath);
548                         }
549                         string fileName = Path.GetFileName(filePath);
550                         string extension = Path.GetExtension(fileName);
551                         if (extension != ".cmp" && extension != ".cmproj" && extension != ".cmm")
552                         {
553                             throw Exception("invalid reference path extension '" + extension + "' (not .cmp, .cmproj or .cmm)");
554                         }
555                         string referencePath = Path.GetDirectoryName(filePath);
556                         if (Path.IsRelative(referencePath))
557                         {
558                             referencePath = Path.Combine(systemLibDir, referencePath);
559                         }
560                         referencePath = Path.Combine(referencePath, Path.ChangeExtension(fileName, ".cmm"));
561                         if (!File.Exists(referencePath))
562                         {
563                             referencePath = Path.GetDirectoryName(referenceDeclaration->FilePath());
564                             if (Path.IsRelative(referencePath))
565                             {
566                                 referencePath = Path.Combine(outDirBasePath, referencePath);
567                             }
568                             referencePath = GetFullPath(Path.ChangeExtension(Path.Combine(Path.Combine(Path.Combine(referencePath, "lib"), config), fileName), ".cmm"));
569                         }
570                         if (Find(references.CBegin(), references.CEnd(), referencePath) == references.CEnd())
571                         {
572                             references.Add(referencePath);
573                         }
574                         break;
575                     }
576                     case ProjectDeclarationType.sourceFileDeclaration:
577                     {
578                         SourceFileDeclaration* sourceFileDeclaration = cast<SourceFileDeclaration*>(declaration.Get());
579                         string filePath = Path.MakeCanonical(sourceFileDeclaration->FilePath());
580                         relativeSourceFilePaths.Add(filePath);
581                         if (Path.IsRelative(filePath))
582                         {
583                             filePath = Path.Combine(sourceBasePath, filePath);
584                         }
585                         string extension = Path.GetExtension(filePath);
586                         if (extension != ".cm")
587                         {
588                             throw Exception("invalid source file extension '" + extension + "' (not .cm)");
589                         }
590                         string sourceFilePath = GetFullPath(filePath);
591                         if (!File.Exists(sourceFilePath))
592                         {
593                             throw Exception("source file path '" + sourceFilePath + "' not found");
594                         }
595                         if (Find(sourceFilePaths.CBegin(), sourceFilePaths.CEnd(), sourceFilePath) == sourceFilePaths.CEnd())
596                         {
597                             sourceFilePaths.Add(sourceFilePath);
598                         }
599                         break;
600                     }
601                     case ProjectDeclarationType.resourceFileDeclaration:
602                     {
603                         ResourceFileDeclaration* resourceFileDeclaration = cast<ResourceFileDeclaration*>(declaration.Get());
604                         string filePath = resourceFileDeclaration->FilePath();
605                         relativeResourceFilePaths.Add(filePath);
606                         if (Path.IsRelative(filePath))
607                         {
608                             filePath = Path.Combine(sourceBasePath, filePath);
609                         }
610                         string extension = Path.GetExtension(filePath);
611                         if (extension != ".xml")
612                         {
613                             throw Exception("invalid resource file extension '" + extension + "' (not .xml)");
614                         }
615                         string resourceFilePath = GetFullPath(filePath);
616                         if (!File.Exists(resourceFilePath))
617                         {
618                             throw Exception("resource file path '" + resourceFilePath + "' not found");
619                         }
620                         if (Find(resourceFilePaths.CBegin(), resourceFilePaths.CEnd(), resourceFilePath) == resourceFilePaths.CEnd())
621                         {
622                             resourceFilePaths.Add(resourceFilePath);
623                         }
624                         break;
625                     }
626                     case ProjectDeclarationType.targetDeclaration:
627                     {
628                         TargetDeclaration* targetDeclaration = cast<TargetDeclaration*>(declaration.Get());
629                         target = targetDeclaration->GetTarget();
630                         break;
631                     }
632                     case ProjectDeclarationType.textFileDeclaration:
633                     {
634                         TextFileDeclaration* textFileDeclaration = cast<TextFileDeclaration*>(declaration.Get());
635                         string filePath = Path.MakeCanonical(textFileDeclaration->FilePath());
636                         relativeTextFilePaths.Add(filePath);
637                         if (Path.IsRelative(filePath))
638                         {
639                             filePath = Path.Combine(sourceBasePath, filePath);
640                         }
641                         string textFilePath = GetFullPath(filePath);
642                         textFilePaths.Add(textFilePath);
643                         break;
644                     }
645                     default:
646                     {
647                         unknown = true;
648                         break;
649                     }
650                 }
651                 if (unknown)
652                 {
653                     throw Exception("unknown project declaration");
654                 }
655             }

688     public nothrow List<Project*> GetReferencedProjects(Project* project, Solution* solution)
689     {
690         List<Project*> referencedProjects;
691         for (const string& referencedProjectFilePath : project->ReferencedProjectFilePaths())
692         {
693             string rpfp = GetFullPath(referencedProjectFilePath);
694             long n = solution->Projects().Count();
695             bool found = false;
696             long i = 0;
697             while (i < n && !found)
698             {
699                 Project* solutionProject = solution->Projects()[i].Get();
700                 string fp = GetFullPath(solutionProject->FilePath());
701                 if (fp == rpfp)
702                 {
703                     referencedProjects.Add(solutionProject);
704                     found = true;
705                 }
706                 ++i;
707             }
708         }
709         return referencedProjects;
710     }
