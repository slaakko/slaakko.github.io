453         public nothrow void Register(NodeType nodeType, NodeCreator* creator)
454         {
455             creators[cast<long>(cast<byte>(nodeType))] = UniquePtr<NodeCreator>(creator);
456         }
457         public Node* CreateNode(NodeType nodeType, const System.Lex.Span& span, const Uuid& moduleId)
458         {
459             const UniquePtr<NodeCreator>& creator = creators[cast<long>(cast<byte>(nodeType))];
460             if (!creator.IsNull())
461             {
462                 Node* value = creator->CreateNode(span, moduleId);
463                 if ((value != null))
464                 {
465                     return value;
466                 }
467                 else
468                 {
469                     throw Exception("could not create node");
470                 }
471             }
472             else
473             {
474                 throw Exception("no creator for node type \'" + NodeTypeStr(nodeType) + "\'");
475             }
476         }
477         private List<UniquePtr<NodeCreator>> creators;
478     }
