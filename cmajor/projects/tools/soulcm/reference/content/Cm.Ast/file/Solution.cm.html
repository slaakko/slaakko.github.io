 16     internal void Visit(List<ustring>& order, const ustring& projectName, HashSet<ustring>& visited, HashSet<ustring>& tempVisit, HashMap<ustring, ProjectDependencyDeclaration*>& dependencyMap, 
 17         Solution* solution)
 18     {
 19         if (tempVisit.CFind(projectName) == tempVisit.CEnd())
 20         {
 21             if (visited.CFind(projectName) == visited.CEnd())
 22             {
 23                 tempVisit.Insert(projectName);
 24                 HashMap<ustring, ProjectDependencyDeclaration*>.ConstIterator it = dependencyMap.CFind(projectName);
 25                 if (it != dependencyMap.CEnd())
 26                 {
 27                     ProjectDependencyDeclaration* dependencyDeclaration = it->second;
 28                     for (const ustring& dependentProject : dependencyDeclaration->DependsOnProjects())
 29                     {
 30                         Visit(order, dependentProject, visited, tempVisit, dependencyMap, solution);
 31                     }
 32                     tempVisit.Remove(projectName);
 33                     visited.Insert(projectName);
 34                     order.Add(projectName);
 35                 }
 36                 else
 37                 {
 38                     throw Exception("project '" + ToUtf8(projectName) + "' not found in dependencies of solution '" + ToUtf8(solution->Name()) + "' (" + GetFullPath(solution->FilePath()) + ")");
 39                 }
 40             }
 41         }
 42         else
 43         {
 44             throw Exception("circular project dependency '" + ToUtf8(projectName) + "' detected in dependencies of solution '" + ToUtf8(solution->Name()) + 
 45                 "' (" + GetFullPath(solution->FilePath()) + ")");
 46         }
 47     }

162         public void ResolveDeclarations()
163         {
164             for (const UniquePtr<SolutionDeclaration>& declaration : declarations)
165             {
166                 bool unknown = false;
167                 switch (declaration->GetDeclarationType())
168                 {
169                     case SolutionDeclarationType.projectDeclaration:
170                     {
171                         SolutionProjectDeclaration* solutionProjectDeclaration = cast<SolutionProjectDeclaration*>(declaration.Get());
172                         string filePath = Path.MakeCanonical(solutionProjectDeclaration->FilePath());
173                         relativeProjectFilePaths.Add(filePath);
174                         if (Path.IsRelative(filePath))
175                         {
176                             filePath = Path.Combine(basePath, filePath);
177                         }
178                         string extension = Path.GetExtension(filePath);
179                         if (extension != ".cmp")
180                         {
181                             throw Exception("invalid project file extension '" + extension + "' (not .cmp)");
182                         }
183                         if (!File.Exists(filePath))
184                         {
185                             throw Exception("project file '" + filePath + "' not found");
186                         }
187                         string projectFilePath = GetFullPath(filePath);
188                         if (Find(projectFilePaths.CBegin(), projectFilePaths.CEnd(), projectFilePath) == projectFilePaths.CEnd())
189                         {
190                             projectFilePaths.Add(projectFilePath);
191                         }
192                         break;
193                     }
194                     case SolutionDeclarationType.activeProjectDeclaration:
195                     {
196                         SolutionActiveProjectDeclaration* activeProjectDeclaration = cast<SolutionActiveProjectDeclaration*>(declaration.Get());
197                         activeProjectName = activeProjectDeclaration->ActiveProjectName();
198                         break;
199                     }
200                     case SolutionDeclarationType.dependencyDeclaration:
201                     {
202                         ProjectDependencyDeclaration* dependencyDeclaration = cast<ProjectDependencyDeclaration*>(declaration.Get());
203                         dependencyMap[dependencyDeclaration->ProjectName()] = dependencyDeclaration;
204                         break;
205                     }
206                     default:
207                     {
208                         unknown = true;
209                         break;
210                     }
211                     if (unknown)
212                     {
213                         throw Exception("unknown solution declaration");
214                     }
215                 }
216             }
217         }
218         public List<Project*> CreateBuildOrder()
219         {
220             AddDependencies();
221             List<Project*> buildOrder;
222             HashMap<ustring, Project*> projectMap;
223             for (const UniquePtr<Project>& project : projects)
224             {
225                 projectMap[project->Name()] = project.Get();
226             }
227             List<ustring> order;
228             HashSet<ustring> visited;
229             HashSet<ustring> tempVisit;
230             for (const UniquePtr<Project>& project : projects)
231             {
232                 if (visited.CFind(project->Name()) == visited.CEnd())
233                 {
234                     Visit(order, project->Name(), visited, tempVisit, dependencyMap, this);
235                 }
236             }
237             for (const ustring& projectName : order)
238             {
239                 HashMap<ustring, Project*>.ConstIterator it = projectMap.CFind(projectName);
240                 if (it != projectMap.CEnd())
241                 {
242                     buildOrder.Add(it->second);
243                 }
244                 else
245                 {
246                     throw Exception("project name '" + ToUtf8(projectName) + "' not found in solution '" + ToUtf8(Name()) + "' (" + GetFullPath(FilePath()) + ")");
247                 }
248             }
249             return buildOrder;
250         }
251         private void AddDependencies()
252         {
253             for (const UniquePtr<Project>& project : projects)
254             {
255                 ProjectDependencyDeclaration* dependencyDeclaration = null;
256                 HashMap<ustring, ProjectDependencyDeclaration*>.ConstIterator it = dependencyMap.CFind(project->Name());
257                 if (it != dependencyMap.CEnd())
258                 {
259                     dependencyDeclaration = it->second;
260                 }
261                 else
262                 {
263                     ProjectDependencyDeclaration* additionalDeclaration = new ProjectDependencyDeclaration(project->Name());
264                     additionalDependencyDeclarations.Add(UniquePtr<ProjectDependencyDeclaration>(additionalDeclaration));
265                     dependencyDeclaration = additionalDeclaration;
266                     dependencyMap[project->Name()] = dependencyDeclaration;
267                 }
268                 for (const UniquePtr<Project>& projectToCheck : projects)
269                 {
270                     if (projectToCheck != project)
271                     {
272                         if (project->DependsOn(projectToCheck.Get()))
273                         {
274                             project->AddDependsOnProject(projectToCheck.Get());
275                             if (Find(dependencyDeclaration->DependsOnProjects().CBegin(), dependencyDeclaration->DependsOnProjects().CEnd(), projectToCheck->Name()) == dependencyDeclaration->DependsOnProjects().CEnd())
276                             {
277                                 dependencyDeclaration->AddDependency(projectToCheck->Name());
278                             }
279                         }
280                     }
281                 }
282             }
