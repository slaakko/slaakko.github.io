 26         {

 30                 System.Security.Sha1 sha1;
 31                 Uuid randomUuid = Uuid.Random();
 32                 for (byte x : randomUuid)
 33                 {
 34                     sha1.Process(x);
 35                 }
 36                 id.Reset(new IdentifierNode(span_, moduleId_, u"unnamed_ns_" + ToUtf32(sha1.GetDigest())));
 37             }
 38             id->SetParent(this);
 39         }

101         public override Node* Clone(CloneContext& cloneContext) const
102         {
103             NamespaceNode* clone = new NamespaceNode(GetSpan(), ModuleId(), cast<IdentifierNode*>(id->Clone(cloneContext)));
104             int n = members.Count();
105             for (int i = 0; i < n; ++i;)
106             {
107                 Node* member = members[i];
108                 if (cloneContext.MakeTestUnits() && member->GetNodeType() == NodeType.functionNode && (cast<FunctionNode*>(member)->GetSpecifiers() & Specifiers.unit_test_) != Specifiers.none)
109                 {
110                     FunctionNode* unitTestFunction = cast<FunctionNode*>(member->Clone(cloneContext));
111                     unitTestFunction->SetParent(this);
112                     cloneContext.AddUnitTestFunction(unitTestFunction);
113                 }
114                 else
115                 {
116                     clone->AddMember(member->Clone(cloneContext));
117                 }
118             }
119             return clone;
120         }
